**距離計算（Haversine公式）**

```javascript
function calcDistance(lat1, lon1, lat2, lon2) {
  const R = 6371000; // 地球の半径（メートル）
  const dLat = (lat2 - lat1) * Math.PI / 180;
  const dLon = (lon2 - lon1) * Math.PI / 180;
  const a = Math.sin(dLat/2) ** 2
    + Math.cos(lat1*Math.PI/180) * Math.cos(lat2*Math.PI/180)
    * Math.sin(dLon/2) ** 2;
  return R * 2 * Math.atan2(Math.sqrt(a), Math.sqrt(1-a));
}
```

**方向計算**

```javascript
function calcAngle(lat1, lon1, lat2, lon2) {
  const dLon = (lon2 - lon1) * Math.PI / 180;
  const y = Math.sin(dLon) * Math.cos(lat2 * Math.PI / 180);
  const x = Math.cos(lat1 * Math.PI / 180) * Math.sin(lat2 * Math.PI / 180)
    - Math.sin(lat1 * Math.PI / 180) * Math.cos(lat2 * Math.PI / 180) * Math.cos(dLon);
  return Math.atan2(y, x); // ラジアン
}
```

**ドット位置計算（ズーム対応）**

```javascript
const ZOOM_RANGES = { wide: 100, mid: 30, close: 10 };

function calcDotPos(distance, angle, zoomMode, radarRadius) {
  const range = ZOOM_RANGES[zoomMode];
  const ratio = Math.min(distance / range, 1.05);
  const r = ratio * radarRadius;
  return {
    x: CENTER_X + r * Math.cos(angle),
    y: CENTER_Y + r * Math.sin(angle),
    inRange: distance <= range * 1.1,
  };
}
```

**音処理（Web Audio API）**

```javascript
function playBeep(distance) {
  const ctx = new AudioContext();
  const osc = ctx.createOscillator();
  const gain = ctx.createGain();
  osc.connect(gain);
  gain.connect(ctx.destination);
  // 距離が近いほど高音・短い間隔
  osc.frequency.value = distance < 10 ? 880 : distance < 30 ? 660 : 440;
  gain.gain.setValueAtTime(0.3, ctx.currentTime);
  gain.gain.exponentialRampToValueAtTime(0.001, ctx.currentTime + 0.1);
  osc.start();
  osc.stop(ctx.currentTime + 0.1);
}
```

### 1.5 認証処理（auth.js）

**WebAuthn（指紋・顔認証）**

```javascript
async function authenticateParent() {
  if (!window.PublicKeyCredential) {
    return authenticateWithPin(); // フォールバック
  }
  try {
    const credential = await navigator.credentials.get({
      publicKey: {
        challenge: new Uint8Array(32),
        timeout: 60000,
        userVerification: 'required',
      }
    });
    return credential !== null;
  } catch {
    return authenticateWithPin(); // 失敗時もPINへ
  }
}
```

**PIN認証**

```javascript
async function authenticateWithPin(inputPin) {
  const result = await api.verifyPin(inputPin);
  return result.verified;
}
```

**認証セッション管理**

```javascript
const AUTH_DURATION_MS = 30 * 60 * 1000; // 30分

function setParentSession() {
  localStorage.setItem('parentAuth', Date.now());
}

function isParentAuthorized() {
  const t = localStorage.getItem('parentAuth');
  return t && (Date.now() - parseInt(t)) < AUTH_DURATION_MS;
}
```

### 2.2 APIエンドポイント詳細

#### GET /api/wishes
* 概要：欲しいものリスト一覧取得（欲しい度降順）
* レスポンス例：
```json
[
  {
    "id": 1,
    "content": "焼肉食べたい！",
    "star_rating": 3,
    "parent_approved": true,
    "created_at": "2026-03-28T10:00:00"
  }
]
```

#### POST /api/wishes
* 概要：お願いごと新規登録
* 5件上限チェック：超える場合は400を返す
* リクエスト：`{ "content": "焼肉食べたい！", "star_rating": 3 }`
* レスポンス：登録したWishオブジェクト

#### PUT /api/wishes/{id}
* 概要：お願いごと更新（内容・欲しい度）
* 存在しないIDの場合：404を返す

#### DELETE /api/wishes/{id}
* 概要：お願いごと削除
* 存在しないIDの場合：404を返す

#### PUT /api/wishes/{id}/approve
* 概要：親が承認
* parent_approved を true に更新

#### PUT /api/wishes/{id}/unapprove
* 概要：承認取り消し
* parent_approved を false に更新

#### GET /api/album
* 概要：アルバム一覧取得（新しい順）
* レスポンス例：
```json
[
  {
    "id": 1,
    "wish_id": 1,
    "wish_content": "焼肉食べたい！",
    "note": "たのしかった！",
    "photo_path": "/uploads/album/1.jpg",
    "fulfilled_at": "2026-03-28T18:00:00"
  }
]
```

#### POST /api/album
* 概要：アルバム記録の保存
* 写真はmultipart/form-dataで受け取りuploads/album/に保存
* リクエスト：`{ "wish_id": 1, "note": "たのしかった！", "photo": <file> }`

#### GET /api/settings
* 概要：設定取得（なければデフォルト値で返す）
* レスポンス例：
```json
{
  "radar_style": "graph",
  "ball_style": "D",
  "ball_count": 7,
  "quest_enabled": true,
  "auth_type": "pin"
}
```

#### PUT /api/settings
* 概要：設定更新

#### POST /api/settings/pin
* 概要：PINコード設定・変更
* リクエスト：`{ "pin": "1234" }`
* bcryptでハッシュ化してDB保存

#### POST /api/settings/pin/verify
* 概要：PINコード検証
* リクエスト：`{ "pin": "1234" }`
* レスポンス：`{ "verified": true }`

### 2.3 モデル定義（models.py）

```python
class Wish(Base):
    __tablename__ = "wishes"

    id              = Column(Integer, primary_key=True, autoincrement=True)
    content         = Column(String(50), nullable=False)
    star_rating     = Column(Integer, nullable=False)  # 1〜3
    parent_approved = Column(Boolean, default=False, nullable=False)
    created_at      = Column(DateTime, default=datetime.utcnow, nullable=False)
    updated_at      = Column(DateTime, default=datetime.utcnow,
                             onupdate=datetime.utcnow, nullable=False)

    album = relationship("Album", back_populates="wish", uselist=False)


class Album(Base):
    __tablename__ = "album"

    id           = Column(Integer, primary_key=True, autoincrement=True)
    wish_id      = Column(Integer, ForeignKey("wishes.id"), nullable=False)
    note         = Column(String(100), nullable=False)
    photo_path   = Column(String(255))
    fulfilled_at = Column(DateTime, default=datetime.utcnow, nullable=False)

    wish = relationship("Wish", back_populates="album")


class Settings(Base):
    __tablename__ = "settings"

    id           = Column(Integer, primary_key=True, default=1)
    radar_style  = Column(String(20), default="graph")   # circle/graph/switch/both
    ball_style   = Column(String(10), default="D")       # A/B/C/D
    ball_count   = Column(Integer, default=7)            # 3/5/7
    quest_enabled= Column(Boolean, default=True)
    auth_type    = Column(String(20), default="pin")     # webauthn/pin
    pin_hash     = Column(String(255))
    updated_at   = Column(DateTime, default=datetime.utcnow,
                          onupdate=datetime.utcnow, nullable=False)
```

## 4. UI/UX設計方針

### 4.1 基本方針

**「公園で片手で操作できる」を最優先とする。**

* ボタンは大きく・タップしやすく（最小48px）
* 文字は大きめ（16px以上）
* ダークテーマ（屋外の日差しでも見やすい）
* 操作の迷いをなくすためシンプルな導線

## 5. エラーハンドリング方針

| HTTPステータス | 場面 |
| --- | --- |
| 200 | 正常 |
| 201 | 登録成功 |
| 400 | 業務エラー（5件上限超え・不正な値） |
| 401 | PIN認証失敗 |
| 404 | 指定IDが存在しない |
| 422 | バリデーションエラー |
| 500 | サーバーエラー |

**GPS関連エラー**

| エラーコード | 内容 | 対応 |
| --- | --- | --- |
| PERMISSION_DENIED | 位置情報が拒否された | 許可を促すメッセージ表示 |
| POSITION_UNAVAILABLE | 位置情報が取得できない | リトライボタンを表示 |
| TIMEOUT | タイムアウト | 自動リトライ（3回まで） |
